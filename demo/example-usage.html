<script>
  ShieldScan.protect({
    action:        'warn',
    interval:      1000,
    minConfidence: 40,
    report:        true,          // sends to demo/index.html
    debuggerTrap:  false,         // set true for stronger protection
    onDetected: function(result) {
      reportThreat(result.name);  // manual report shorthand
      console.log('[THREAT]', result);
    }
  });
</script>
